  <script>
    /***********************************************
     * Editable area: change reviews array or timing
     ***********************************************/
    const AUTOPLAY_MS = 3000; // change speed
    const TRANSITION_MS = 450; // should match CSS --transition duration
    const VISIBLE = 3;         // number of cards visible on wide screens

    // Add/Remove/Modify reviews here. Each object becomes a card.
    const reviews = [
      { name: 'Ayesha Khan', role: 'Product Designer', text: 'This product changed how I work — clean, fast and ridiculously reliable. Highly recommended!', avatar: 'A' },
      { name: 'Bilal Raza', role: 'Frontend Dev', text: 'I love the simplicity. Setup took minutes and the results look very professional.', avatar: 'B' },
      { name: 'Sara Ahmed', role: 'Content Creator', text: 'Amazing UX and the animations are buttery smooth. My users love the new review section.', avatar: 'S' },
      { name: 'Hamza Ali', role: 'Entrepreneur', text: 'Great performance and looks great on mobile. The auto-scroll feels natural.', avatar: 'H' },
      { name: 'Zara Iqbal', role: 'Marketer', text: 'Clean, editable and modern — exactly what I needed for our landing page.', avatar: 'Z' }
    ];

    /***********************************************
     * Implementation (you can edit below too if needed)
     ***********************************************/

    const viewport = document.getElementById('viewport');
    const track = document.getElementById('track');
    const dotsWrap = document.getElementById('dots');
    const prevBtn = document.querySelector('.prev');
    const nextBtn = document.querySelector('.next');

    // create slides (including clones for smooth looping)
    const cloneCount = VISIBLE; // clone this many from start/end

    function buildSlides(){
      track.innerHTML = '';
      // clones from end (prepend)
      for(let i = reviews.length - cloneCount; i < reviews.length; i++){
        const r = reviews[(i+reviews.length)%reviews.length];
        track.appendChild(makeSlide(r, true));
      }
      // real slides
      for(let i=0;i<reviews.length;i++) track.appendChild(makeSlide(reviews[i], false));
      // clones from start (append)
      for(let i=0;i<cloneCount;i++) track.appendChild(makeSlide(reviews[i%reviews.length], true));
    }

    function makeSlide(data, isClone){
      const el = document.createElement('article');
      el.className = 'slide';
      if(isClone) el.setAttribute('data-clone','true');
      el.innerHTML = `
        <div class="meta">
          <div class="avatar">${escapeText(data.avatar || data.name.charAt(0))}</div>
          <div class="who">
            <div class="name">${escapeText(data.name)}</div>
            <div class="role">${escapeText(data.role)}</div>
          </div>
          <div class="rating" aria-hidden="true">
            <div class="stars">
              <span class="star">★</span><span class="star">★</span><span class="star">★</span><span class="star">★</span><span class="star">★</span>
            </div>
          </div>
        </div>
        <div class="text">${escapeText(data.text)}</div>
      `;
      return el;
    }

    function escapeText(s){return String(s).replaceAll('&','&amp;').replaceAll('<','&lt;').replaceAll('>','&gt;')}

    buildSlides();

    let slides = Array.from(track.children);
    const total = slides.length; // includes clones
    const realCount = total - cloneCount*2;
    let index = cloneCount; // start centered on first real item (index in 'slides')

    // utility to read gap in px
    function gapPx(){
      const g = getComputedStyle(track).gap || getComputedStyle(track).getPropertyValue('gap');
      return parseFloat(g) || 0;
    }

    function updateLayout(withTransition = true){
      // recompute sizes
      slides = Array.from(track.children);
      const viewportW = viewport.clientWidth;
      const slideW = slides[0].offsetWidth;
      const step = slideW + gapPx();

      // center formula: translateX = (viewportWidth/2) - (slideWidth/2) - index*step
      const translateX = (viewportW/2) - (slideW/2) - index * step;
      if(!withTransition) track.style.transition = 'none';
      track.style.transform = `translateX(${translateX}px)`;
      if(!withTransition){
        // force reflow
        track.getBoundingClientRect();
        track.style.transition = '';
      }

      // assign visual states (center, side, other)
      slides.forEach((s,i)=>{
        s.classList.remove('slide--center','slide--side','slide--other');
        const raw = i - index;
        // normalize raw to shortest circular distance
        let diff = raw;
        const half = Math.floor(total/2);
        if(diff > half) diff -= total;
        if(diff < -half) diff += total;
        if(diff === 0) s.classList.add('slide--center');
        else if(diff === -1 || diff === 1) s.classList.add('slide--side');
        else s.classList.add('slide--other');
      });

      updateDots();
    }

    function next(){ moveTo(index + 1); }
    function prev(){ moveTo(index - 1); }

    let isAnimating = false;

    function moveTo(target){
      if(isAnimating) return;
      isAnimating = true;
      index = target;
      updateLayout(true);
      // after transition, check clones
      window.setTimeout(()=>{
        if(index < cloneCount){
          index = index + realCount; // jumped to left clones -> move to matching real
          updateLayout(false);
        } else if(index >= realCount + cloneCount){
          index = index - realCount; // jumped to right clones
          updateLayout(false);
        }
        isAnimating = false;
      }, TRANSITION_MS);
    }

    // Dots
    function buildDots(){
      dotsWrap.innerHTML = '';
      for(let i=0;i<realCount;i++){
        const b = document.createElement('button');
        b.className = 'dot';
        b.setAttribute('aria-label', `Go to review ${i+1}`);
        b.addEventListener('click', ()=> goTo(i));
        dotsWrap.appendChild(b);
      }
    }

    function updateDots(){
      const logical = normalizeIndex(index);
      Array.from(dotsWrap.children).forEach((d,i)=> d.setAttribute('aria-current', i===logical ? 'true' : 'false'));
    }

    function normalizeIndex(i){
      let r = i - cloneCount; // 0..realCount-1 maybe out of range
      r = ((r % realCount) + realCount) % realCount; // wrap
      return r;
    }

    function goTo(real0){
      if(isAnimating) return;
      index = real0 + cloneCount;
      moveTo(index);
    }

    // autoplay
    let timer = null;
    function startAutoplay(){ stopAutoplay(); timer = setInterval(()=> next(), AUTOPLAY_MS); }
    function stopAutoplay(){ if(timer) clearInterval(timer); timer = null; }

    // pause on hover/focus
    viewport.addEventListener('mouseenter', stopAutoplay);
    viewport.addEventListener('mouseleave', startAutoplay);
    viewport.addEventListener('focusin', stopAutoplay);
    viewport.addEventListener('focusout', startAutoplay);

    // keyboard
    viewport.tabIndex = 0;
    viewport.addEventListener('keydown', (e)=>{ if(e.key==='ArrowRight') next(); if(e.key==='ArrowLeft') prev(); });

    // pointer/drag (simple)
    let drag = {active:false,startX:0,x:0};
    function onDown(px){ drag.active=true; drag.startX=px; drag.x=px; stopAutoplay(); track.style.transition='none'; }
    function onMove(px){ if(!drag.active) return; drag.x = px; const delta = drag.x - drag.startX; track.style.transform = `translateX(${(viewport.clientWidth/2) - (slides[0].offsetWidth/2) - index*(slides[0].offsetWidth + gapPx()) + delta}px)`; }
    function onUp(){ if(!drag.active) return; track.style.transition=''; const delta = drag.x - drag.startX; const threshold = slides[0].offsetWidth * 0.15; if(delta > threshold) prev(); else if(delta < -threshold) next(); else updateLayout(true); drag.active=false; startAutoplay(); }

    viewport.addEventListener('pointerdown', e=> onDown(e.clientX));
    window.addEventListener('pointermove', e=> onMove(e.clientX));
    window.addEventListener('pointerup', onUp);
    viewport.addEventListener('touchstart', e=> onDown(e.touches[0].clientX), {passive:true});

    // buttons
    prevBtn.addEventListener('click', prev);
    nextBtn.addEventListener('click', next);

    // init
    buildDots();
    updateLayout(false);
    startAutoplay();

    // responsive: recalc on resize
    let resizeTimer = null;
    window.addEventListener('resize', ()=>{
      if(resizeTimer) clearTimeout(resizeTimer);
      resizeTimer = setTimeout(()=> updateLayout(false), 120);
    });

  </script>
